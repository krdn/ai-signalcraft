```typescript
export interface AnalysisModule<T = unknown> {
  readonly name: string;
  readonly displayName: string;
  readonly provider: AIProvider;
  readonly model: string;
  readonly schema: z.ZodType<T>;
  buildPrompt(data: AnalysisInput): string;
  buildSystemPrompt(): string;
  buildPromptWithContext?(data: AnalysisInput, priorResults: Record<string, unknown>): string;
}

export interface AnalysisInput {
  jobId: number;
  keyword: string;
  articles: Array<{ title: string; content: string | null; publisher: string | null; publishedAt: Date | null; source: string; }>;
  videos: Array<{ title: string; description: string | null; channelTitle: string | null; viewCount: number | null; likeCount: number | null; publishedAt: Date | null; }>;
  comments: Array<{ content: string; source: string; author: string | null; likeCount: number | null; dislikeCount: number | null; publishedAt: Date | null; }>;
  dateRange: { start: Date; end: Date };
}

export const MODULE_MODEL_MAP: Record<string, { provider: AIProvider; model: string }>;
export const MODULE_NAMES;
```
