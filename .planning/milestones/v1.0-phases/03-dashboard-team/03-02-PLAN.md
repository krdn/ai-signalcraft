2. apps/web/src/server/trpc/routers/pipeline.ts:
```typescript
import { z } from 'zod';
import { protectedProcedure, router } from '../init';
import { db, collectionJobs, analysisResults, analysisReports } from '@ai-signalcraft/core';
import { eq, and, inArray } from 'drizzle-orm';
import { TRPCError } from '@trpc/server';

export const pipelineRouter = router({
  getStatus: protectedProcedure
    .input(z.object({ jobId: z.number() }))
    .query(async ({ input, ctx }) => {
      // 1. collectionJobs에서 수집 상태 조회
      const [job] = await ctx.db.select()
        .from(collectionJobs)
        .where(eq(collectionJobs.id, input.jobId));
      if (!job) throw new TRPCError({ code: 'NOT_FOUND' });

      // 2. 4단계 파이프라인 상태 파생 (per D-12):
      //    수집(collection) -> 정규화(normalization) -> 분석(analysis) -> 리포트(report)
      //
      //    수집 상태: collectionJobs.status
      //      - 'pending' | 'running' = 수집 진행 중
      //      - 'completed' | 'partial_failure' = 수집 완료
      //      - 'failed' = 수집 실패
      //
      //    정규화 상태: 수집 완료 시 자동 처리됨 (BullMQ Flow 내부)
      //      - collectionJobs.status가 'completed'|'partial_failure'이면 정규화 완료로 간주
      //
      //    분석 상태: analysisResults 테이블 조회
      //      - analysisResults WHERE jobId = input.jobId 레코드 존재 여부 + status 확인
      //      - 레코드 없음 = 분석 미시작 (수집/정규화 단계)
      //      - 하나라도 'running' | 'pending' = 분석 진행 중
      //      - 모두 'completed' | 'failed' = 분석 완료
      //
      //    리포트 상태: analysisReports 테이블 조회
      //      - analysisReports WHERE jobId = input.jobId 레코드 존재 = 리포트 완료
      //      - 레코드 없음 + 분석 완료 = 리포트 생성 중
      //
      const analysisRows = await ctx.db.select()
        .from(analysisResults)
        .where(eq(analysisResults.jobId, input.jobId));

      const [report] = await ctx.db.select({ id: analysisReports.id })
        .from(analysisReports)
        .where(eq(analysisReports.jobId, input.jobId))
        .limit(1);

      // 파이프라인 단계 계산
      const collectionDone = ['completed', 'partial_failure'].includes(job.status);
      const collectionFailed = job.status === 'failed';
      const normalizationDone = collectionDone; // BullMQ Flow에서 수집 후 자동 정규화
      const analysisStarted = analysisRows.length > 0;
      const analysisInProgress = analysisRows.some(r => r.status === 'running' || r.status === 'pending');
      const analysisDone = analysisStarted && !analysisInProgress;
      const reportDone = !!report;

      const pipelineStages = {
        collection: { status: collectionFailed ? 'failed' : collectionDone ? 'completed' : job.status === 'running' ? 'running' : 'pending' },
        normalization: { status: collectionFailed ? 'skipped' : normalizationDone ? 'completed' : 'pending' },
        analysis: { status: collectionFailed ? 'skipped' : analysisDone ? 'completed' : analysisInProgress ? 'running' : analysisStarted ? 'running' : 'pending' },
        report: { status: collectionFailed ? 'skipped' : reportDone ? 'completed' : analysisDone ? 'running' : 'pending' },
      };

      return {
        status: job.status,
        progress: job.progress,
        errorDetails: job.errorDetails,
        keyword: job.keyword,
        pipelineStages,
        analysisModuleCount: { total: analysisRows.length, completed: analysisRows.filter(r => r.status === 'completed').length },
        hasReport: reportDone,
      };
    }),
});
````

3. apps/web/src/server/trpc/routers/history.ts:

```typescript
import { z } from 'zod';
import { protectedProcedure, router } from '../init';
import { db, collectionJobs } from '@ai-signalcraft/core';
import { desc, sql } from 'drizzle-orm';

export const historyRouter = router({
  list: protectedProcedure
    .input(
      z.object({
        page: z.number().min(1).default(1),
        perPage: z.number().min(1).max(50).default(20),
      }),
    )
    .query(async ({ input, ctx }) => {
      const offset = (input.page - 1) * input.perPage;
      const jobs = await ctx.db
        .select()
        .from(collectionJobs)
        .orderBy(desc(collectionJobs.createdAt))
        .limit(input.perPage)
        .offset(offset);
      const [{ count }] = await ctx.db.select({ count: sql`count(*)::int` }).from(collectionJobs);
      return { items: jobs, total: count, page: input.page, perPage: input.perPage };
    }),
});
```
