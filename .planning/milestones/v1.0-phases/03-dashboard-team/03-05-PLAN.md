id: integer('id').primaryKey().generatedAlwaysAsIdentity(),
name: text('name').notNull(),
createdBy: text('created_by').references(() => users.id).notNull(),

import { Resend } from 'resend';

const resend = new Resend(process.env.RESEND_API_KEY);

export async function sendInviteEmail(params: {
to: string;
inviterName: string;
teamName: string;
inviteUrl: string;
role: 'admin' | 'member';
}) {
await resend.emails.send({
from: 'AI SignalCraft <noreply@yourdomain.com>', // .env에서 설정 가능하게
to: params.to,
subject: '[AI SignalCraft] 팀 초대',
html: `<p>${params.inviterName}님이 ${params.teamName} 팀에 초대했습니다.</p>
           <p>역할: ${params.role === 'admin' ? '관리자' : '멤버'}</p>
           <a href="${params.inviteUrl}">초대 수락하기</a>
           <p>이 링크는 7일 후 만료됩니다.</p>`,
});
}
