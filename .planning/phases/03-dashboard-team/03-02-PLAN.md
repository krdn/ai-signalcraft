1. apps/web/src/server/trpc/routers/analysis.ts:
```typescript
import { z } from 'zod';
import { protectedProcedure, router } from '../init';
import { db, collectionJobs, triggerCollection } from '@ai-signalcraft/core';
import { eq, desc } from 'drizzle-orm';

export const analysisRouter = router({
  trigger: protectedProcedure
    .input(z.object({
      keyword: z.string().min(1).max(50),
      sources: z.array(z.enum(['naver', 'youtube'])).min(1),
      startDate: z.string(), // ISO date string
      endDate: z.string(),
    }))
    .mutation(async ({ input, ctx }) => {
      // 1. collectionJobs 레코드 생성
      const [job] = await ctx.db.insert(collectionJobs).values({
        keyword: input.keyword,
        startDate: new Date(input.startDate),
        endDate: new Date(input.endDate),
        status: 'pending',
        teamId: /* ctx.session에서 teamId 추출 */ null,
      }).returning();
      // 2. BullMQ 트리거
      await triggerCollection({
        keyword: input.keyword,
        sources: input.sources,
        startDate: input.startDate,
        endDate: input.endDate,
      }, job.id);
      return { jobId: job.id };
    }),

  getResults: protectedProcedure
    .input(z.object({ jobId: z.number() }))
    .query(async ({ input, ctx }) => {
      // analysisResults에서 해당 jobId의 모든 모듈 결과 조회
      const results = await ctx.db.select()
        .from(analysisResults)
        .where(eq(analysisResults.jobId, input.jobId));
      return results;
    }),
});
```

2. apps/web/src/server/trpc/routers/pipeline.ts:
```typescript
export const pipelineRouter = router({
  getStatus: protectedProcedure
    .input(z.object({ jobId: z.number() }))
    .query(async ({ input, ctx }) => {
      const [job] = await ctx.db.select()
        .from(collectionJobs)
        .where(eq(collectionJobs.id, input.jobId));
      if (!job) throw new TRPCError({ code: 'NOT_FOUND' });
      return {
        status: job.status,
        progress: job.progress,
        errorDetails: job.errorDetails,
        keyword: job.keyword,
      };
    }),
});
```

3. apps/web/src/server/trpc/routers/history.ts:
```typescript
export const historyRouter = router({
  list: protectedProcedure
    .input(z.object({
      page: z.number().min(1).default(1),
      perPage: z.number().min(1).max(50).default(20),
    }))
    .query(async ({ input, ctx }) => {
      const offset = (input.page - 1) * input.perPage;
      const jobs = await ctx.db.select()
        .from(collectionJobs)
        .orderBy(desc(collectionJobs.createdAt))
        .limit(input.perPage)
        .offset(offset);
      const [{ count }] = await ctx.db.select({ count: sql`count(*)::int` })
        .from(collectionJobs);
      return { items: jobs, total: count, page: input.page, perPage: input.perPage };
    }),
});
```
