🏗️ 수집 인터페이스 (Collector)
interface Collector<T = unknown> {
  readonly source: string;  // 'naver-news', 'youtube', 'dcinside' 등
  collect(options: CollectionOptions): AsyncGenerator<T[], void, unknown>;
  //                                  ↑ 청크 단위로 yield → 메모리 효율 + 진행률 추적 가능
}

interface CollectionOptions {
  keyword: string;       // 검색 키워드
  startDate?: Date;      // 수집 시작일
  endDate?: Date;        // 수집 종료일
  maxItems?: number;     // 기사/영상 최대 수집량
  maxComments?: number;  // 댓글 최대 수집량
}

🔍 Upsert 전략
-- articles (videos, comments 동일 패턴)
INSERT INTO articles (source, source_id, url, title, content, collected_at, ...)
VALUES ('naver-news', '0123_4567', 'https://...', '제목', '본문', NOW(), ...)
ON CONFLICT (source, source_id)
DO UPDATE SET
  title        = EXCLUDED.title,
  content      = EXCLUDED.content,
  raw_data     = EXCLUDED.raw_data,
  collected_at = NOW();
  -- 중복 URL이 다른 job에서 다시 수집되면 최신 데이터로 갱신
  -- FK는 변경하지 않음 (original job 유지)

기사: 라운드로빈 (시간순 보존)
댓글: 균등분배 (좋아요순 유지)
영상: 균등분배 
// 20,000자 이하 → 단일 API 호출
if (totalChars < 20000) {
  return module.buildPrompt(input);
}

// 20,000자 초과 → 청크 분할
const chunks = splitIntoChunks(input, 15000);

// Map: 청크별 병렬 분석
const mapResults = await Promise.all(
  chunks.map(chunk =>
    analyzeText(prompt, { timeoutMs: 300_000 })
  )
);

// Reduce: 결과 통합
const finalResult = await reduceResults(mapResults);

-- 시맨틱 유사도 검색
SELECT id, title,
  1 - (embedding <=> $1::vector) AS similarity
FROM articles
WHERE embedding IS NOT NULL
ORDER BY embedding <=> $1::vector
LIMIT 20;
