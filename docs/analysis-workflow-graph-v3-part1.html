🏗️ 수집 인터페이스 (Collector)
interface Collector<T = unknown> {
  readonly source: string;           // 'naver-news', 'youtube', 'dcinside' 등
  collect(options: CollectionOptions): AsyncGenerator<T[], void, unknown>;
                         // ↑ 청크 단위로 yield → 메모리 효율 + 진행률 추적 가능
}

interface CollectionOptions {
  keyword: string;        // 검색 키워드
  startDate?: Date;       // 수집 시작일
  endDate?: Date;         // 수집 종료일
  maxItems?: number;      // 기사/영상 최대 수집량
  maxComments?: number;   // 댓글 최대 수집량
}

🔀 Map-Reduce 청킹 — 대량 데이터 처리
청킹 기준값
chunkTargetChars15,000자 (≈4K토큰)
minCharsForChunking20,000자 이상만 청킹
mapTimeoutMs300,000ms (5분)
reduceTimeoutMs300,000ms (5분)
분배 전략
기사: 라운드로빈 (시간순 보존)
댓글: 균등분배 (좋아요순 유지)
영상: 균등분배 
// 20,000자 이하 → 청킹 없이 단일 API 호출
if (totalChars < 20000) {
  return module.buildPrompt(input); // 단일 호출
}

// 20,000자 초과 → 청크 분할
const chunks = splitIntoChunks(input, 15000);

// Map 단계: 청크별 병렬 분석
const mapResults = await Promise.all(
  chunks.map(chunk => analyzeText(prompt, { timeoutMs: 300_000 }))
);

// Reduce 단계: 청크 결과 통합
const finalResult = await reduceResults(mapResults);
